\begin{abstract}
Michael Polanyi's tacit knowledge—``we know more than we can tell''—is formalized in the Spherepop calculus as \emph{cryptographic boundary entropy}: high-dimensional, non-compressible semantic metadata bound to embodied, pre-articulable interiors. In the Playcosm, tacit knowledge manifests in gestural syntax (pushing a toy car encodes momentum) and prefigurative affordances (gliders debug flight intuition). We prove that spheres with tacit entropy $h \geq h_0$ are anti-admissible against compressive pop regimes, resisting Goodhartian flattening and shallow gamification. The synthesis yields a cryptographic-resistance theorem for epistemic sovereignty in unified play.
\end{abstract}

\section{Polanyi's Tacit Knowledge: Phenomenological Core}

Polanyi (1966) asserts:
\begin{itemize}
    \item \textbf{Subsidiary awareness}: Knowledge dwells in particulars (gestures, tools) attended \emph{from}, not \emph{to}.
    \item \textbf{Focal integration}: Meaning emerges in action, not explicit proposition.
    \item \textbf{Inarticulability}: Tacit coefficients cannot be fully specified without destroying function.
\end{itemize}

Example: Riding a bicycle—muscle tensions, balance intuitions—cannot be exhaustively codified.

\section{Spherepop Formalization}

\begin{definition}[Tacit Sphere]
A sphere $S^\text{tacit} = (I, B, \Sigma_\text{tacit})$ where:
\begin{itemize}
    \item $I$: embodied, pre-articulate interior (gestures, intuitions),
    \item $B$: observable interface (explicit rules, metrics),
    \item $\Sigma_\text{tacit}: I \to B$ is \emph{partial, lossy, high-entropy}.
\end{itemize}
\end{definition}

\begin{definition}[Tacit Entropy]
\[
h_\text{tacit}(S) = H(I \mid B) = H(I) - I(I;B),
\]
conditional entropy of interior given boundary. High $h_\text{tacit}$ implies inarticulable residue.
\end{definition}

\begin{axiom}[Polanyi Compression Barrier]
Any explicit encoding $B' \supseteq B$ satisfies:
\[
h_\text{tacit}(S) \leq H(I \mid B') + \text{loss}(o),
\]
where $o$ is functional objective (e.g., flight stability). Full articulation destroys $o$.
\end{axiom}

\section{Anti-Admissibility via Tacit Entropy}

\begin{theorem}[Tacit Cryptographic Resistance]
Let $S^\text{tacit}$ have $h_\text{tacit} \geq h_0 = \log_2 (q_\text{max} / c_\text{query}) + \log_2 d$, with ritual duration $d \geq d_0$. Then $S^\text{tacit}$ is anti-admissible w.r.t. any compressive regime $\mathcal{R}$:
\[
\Pr[\text{pop}(S^\text{tacit}, T) \text{ succeeds}] \leq 2^{-h_\text{tacit}}.
\]
\end{theorem}

\begin{proof}
\textbf{Phase 1: Ritual Gating.} Valid transfer requires $d$ sequential embodied steps. Expected trials: $(1/\delta)^d > t_\text{max}/c_\text{step}$.

\textbf{Phase 2: Tacit Revelation.} Post-ritual, $k$ is partially exposed, but full $h_\text{tacit}$ requires reconstruction of $I$. Query complexity: $\Omega(2^{h_\text{tacit}})$.

\textbf{Phase 3: Compression Loss.} Any merge $M$ with $H_\text{boundary}(M) < h_\text{tacit}$ loses $o$ (Polanyi barrier). Cost:
\[
-\lambda H_\text{boundary}(M) < -\lambda h_\text{tacit} \ll -\tau.
\]
Pop undefined. Joint probability: exponential in $h_\text{tacit} + d$.
\qed
\end{proof}

\begin{corollary}
Shallow gamification (low $h_\text{tacit}$ metrics) cannot assimilate prefigurative play (high $h_\text{tacit}$). Goodhart collapse fails.
\end{corollary}
